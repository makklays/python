

1. Переменные: динамические и изменяемые

// В Python нет ключевых слов val, var или final. Вы просто пишете имя и присваиваете значение. Переменную можно изменить в любой момент.
// Константы в Python существуют только «на словах»: их принято называть ЗАГЛАВНЫМИ БУКВАМИ, чтобы другие программисты их не трогали.
name = "Иван"  # Обычная переменная
age = 25       # Обычная переменная

PI = 3.14159   # Так обозначают константы (но технически ее можно перезаписать)

2. Структура кода (отступы)

// Блоки кода выделяются двоеточием : и обязательным отступом в 4 пробела. Круглые скобки в условиях if не нужны. Вместо else if пишется сокращение elif.
if age >= 18:
    print("Доступ разрешен")
else:
    print("Доступ запрещен")

3. Match (аналог switch)

// Появился в Python 3.10. Он работает точно так же, как в Scala. Роль дефолтного варианта (default или _) выполняет одиночный знак подчеркивания _.
day = 1
match day:
    case 1:
        day_name = "Monday"
    case 2:
        day_name = "Tuesday"
    case _:
        day_name = "Unknown"

4. Функции (методы)

// Функции создаются с помощью ключевого слова def. Результат возвращается через явное слово return. Если его не написать, функция вернет пустоту (None).
def sum_numbers(x, y):
    return x + y
# Вызов функции
result = sum_numbers(5, 10) # result равен 15

5. Lambda (анонимная функция)

// Лямбды в Python пишутся в одну строку с помощью ключевого слова lambda. Они могут содержать только одно выражение (внутри нельзя писать сложные циклы или условия).
add = lambda a, b: a + b

# Использование в коллекциях (через встроенную функцию map):
numbers = [1, 2, 3]
doubled = list(map(lambda x: x * 2, numbers)) # Получим [2, 4, 6]

6. Создание объектов и классов

// Вместо перегруженных конструкторов используется специальный метод __init__. Первым аргументом в методах класса всегда должен идти self (ссылка на текущий объект). Ключевое слово new не используется.
class Car:
    def __init__(self, brand, speed):
        self.brand = brand  # Поле класса
        self.speed = speed  # Поле класса

    def drive(self):
        print(f"Машина {self.brand} едет со скоростью {self.speed} км/ч")

# Создаем машину
my_car = Car("Toyota", 90)
my_car.drive()

7. Dataclass (аналог Case class в Scala / Record в Java)

// Для быстрого создания классов с данными в Python есть специальный декоратор @dataclass. Он сам создаст конструктор и метод красивого вывода класса на экран.
from dataclasses import dataclass

@dataclass
class User:
    id: int
    name: str
    
# Создание:
user = User(1, "Alex")

8. List Comprehensions (Генерация коллекций «на лету»)

// Вместо методов .map() и .filter() в Python чаще используется невероятно удобный синтаксис генераторов списков (очень похоже на for ... yield в Scala).
numbers = [1, 2, 3, 4, 5]
# Аналог map: умножить все на 2
doubled = [x * 2 for x in numbers] # [2, 4, 6, 8, 10]
# Аналог filter: оставить только больше 1
filtered = [x for x in numbers if x > 1] # [2, 3, 4, 5]

9. Циклы

// Цикл for работает как итератор по коллекциям (через ключевое слово in). Для генерации последовательности чисел используется функция range().
# Перебор элементов списка
for n in [1, 2, 3]:
    print(n)
    
# Цикл от 0 до 4 включительно
for i in range(5):
    print(i)

10. Эквивалент Enum

// Перечисления не являются встроенной частью базового синтаксиса, но их можно импортировать из стандартной библиотеки.
from enum import Enum

class Status(Enum):
    ACTIVE = 1
    BLOCKED = 2
    DELETED = 3

# Использование:
status = Status.ACTIVE

11. Абстрактные классы (аналог интерфейсов)

// В Python нет ключевого слова interface или trait. Используется множественное наследование и специальный модуль abc.
from abc import ABC, abstractmethod

class Animal(ABC):
    @abstractmethod
    def sound(self):
        pass

class Dog(Animal):
    def sound(self):
        return "Woof"

12. Статические методы (@staticmethod)

// Если методу внутри класса не нужны данные объекта (не нужен self), перед ним ставится декоратор @staticmethod. Это аналог static в Java или методов внутри object в Scala.
class Utils:
    @staticmethod
    def hello():
        print("Hello")

# Вызов без создания объекта класса:
Utils.hello()

13. Форматирование строк (f-строки)

// Самый современный способ подставлять переменные в текст — поставить букву f перед кавычками. Переменные пишутся внутри фигурных скобок { }.
name = "Alex"
greeting = f"Hello, {name}!"
print(greeting) # Выведет: Hello, Alex!

14. Null-безопасность (None)

// Аналогом null в Python является None. Строгой встроенной защиты от него на этапе компиляции нет (так как нет компилятора), поэтому проверки делаются через оператор is.
name = None

if name is None:
    print("Empty")
else:
    print(name)

15. Основные коллекции

[1, 2, 3]               # Список (List) — изменяемый, аналог ArrayList
(1, 2, 3)               # Кортеж (Tuple) — неизменяемый список
{1, 2, 3}               # Множество unique элементов (Set)
{"a": 1, "b": 2}        # Словарь / Карта (Dictionary / Dict)
